2.1 Making some charts outside of the school
<div id='chart'></div>
var evenNumbers = [2, 4, 6, 8, 10]; var w = 600; var h = 100; var root = d3.select('#chart').append('svg') .attr('width', w) .attr('height', h) .style('border', '1px solid black'); var barHeight = h / evenNumbers.length; root.selectAll('rect.number') .data(evenNumbers).enter() .append('rect') .attr('class', 'number') .attr('x', 0) .attr('y', function(d, i) { return i * barHeight; }) .attr('width', function(d) { return 60 * d; }) .attr('height', barHeight) .attr('fill', '#cff') .attr('stroke', '#666');
